class Pilha:
    """Implementação didática de uma estrutura de dados Pilha."""
    
    def __init__(self):
        # Inicializa uma lista vazia para armazenar os elementos
        self.__itens = []

    def esta_vazia(self):
        """Retorna True se a pilha não contiver elementos."""
        return len(self.__itens) == 0

    def empilhar(self, item):
        """Adiciona um elemento ao topo da pilha (Push)."""
        self.__itens.append(item)

    def desempilhar(self):
        """Remove e retorna o elemento do topo (Pop)."""
        if self.esta_vazia():
            raise IndexError("Atenção: A pilha está vazia!")
        return self.__itens.pop()

    def ver_topo(self):
        """Consulta o elemento do topo sem remover."""
        if not self.esta_vazia():
            return self.__itens[-1]
        return None
    def listar(self):
        """Exibe todos os elementos da pilha do topo para a base."""
        if self.esta_vazia():
            print("A pilha está vazia.")
            return
        
        print("Elementos na pilha (topo -> base):")
        # Inverte a lista para mostrar o topo primeiro
        for item in reversed(self.__itens):
            print(f"- {item}")

    def tamanho(self):
        """Retorna a quantidade de elementos na pilha."""
        return len(self.__itens)
    def esvaziar(self):
        """Remove todos os elementos da pilha de uma vez."""
        self.__itens = []
        print("A pilha foi totalmente esvaziada.")


# --- Exemplo de Teste ---
if __name__ == "__main__":
    
    minha_pilha = Pilha()
    minha_pilha.listar()
    minha_pilha.empilhar("Livro A")
    minha_pilha.empilhar("Livro B")
    minha_pilha.empilhar("Livro C")
    minha_pilha.empilhar("Livro D")
    minha_pilha.listar()
    print(f"Topo atual: {minha_pilha.ver_topo()}") 
    print(f"Removendo: {minha_pilha.desempilhar()}")
    minha_pilha.listar()
    print(f"Tamanho: {minha_pilha.tamanho()}")
    
      # Chama o novo método
    minha_pilha.esvaziar()
    
    print(f"Tamanho após limpar: {minha_pilha.tamanho()}") # Retornará 0
